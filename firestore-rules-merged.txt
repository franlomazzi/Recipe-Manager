rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper function to check authentication
    function isAuthenticated() {
      return request.auth != null;
    }

    // Helper function to check if this is the Bridge app's hardcoded userId
    function isBridgeAppUser(userId) {
      return userId == "0CJxUFHYFwgNdG1hHtZbHMGWvvS2"
          || userId == "mbkgRXpOZ6ZjY4bbUdzAwVTeB8D2";
    }

    // Rules for Workout App User Profile & Settings
    match /users/{userId}/{allPaths=**} {
      allow read, write: if isAuthenticated() && request.auth.uid == userId;
    }
    // -------------------------------------------------------------------------
    // Weight Tracker Collections
    // -------------------------------------------------------------------------

    match /weight_settings/{userId} {
      allow read, write: if isAuthenticated() && request.auth.uid == userId;
    }
    match /weight_daily_logs/{docId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }
    match /weight_weekly_logs/{docId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }
    // -------------------------------------------------------------------------
    // Workout App Collections
    // -------------------------------------------------------------------------

    match /workout_sessions/{sessionId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }

    match /workout_programs/{programId} {
      allow read: if isAuthenticated() && resource.data.ownerUserId == request.auth.uid;
      allow create: if isAuthenticated() && request.resource.data.ownerUserId == request.auth.uid;
      allow update, delete: if isAuthenticated() && resource.data.ownerUserId == request.auth.uid;
    }
    match /workout_exercises/{exerciseId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }

    match /workout_families/{familyId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    match /running_sessions/{sessionId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }

    // Weekly Program Insights
    match /workout_weekly_insights/{insightId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    // -------------------------------------------------------------------------
    // Nutrition App Collections
    // -------------------------------------------------------------------------

    match /nutrition_logs/{docId} {
      allow read, write: if isAuthenticated() && (docId.matches("^" + request.auth.uid + "_.*") || (resource != null && resource.data.userId == request.auth.uid) || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }

    // Shared ingredient library (used by both Recipe Manager and Food Tracking apps)
    // Reads are open to any signed-in user so Recipe Manager can merge two
    // household members' libraries client-side. Writes remain creator-only.
    match /nutrition_ingredients/{docId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated()
        && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAuthenticated()
        && resource.data.userId == request.auth.uid;
    }

    // Shared nutrition_meals collection (used by both Recipe Manager and Food Tracking apps)
    // Reads: own docs OR docs flagged householdShared. Writes: creator-only.
    match /nutrition_meals/{mealId} {
      allow read: if isAuthenticated()
        && (resource == null
            || resource.data.userId == request.auth.uid
            || resource.data.householdShared == true);
      allow create: if isAuthenticated()
        && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAuthenticated()
        && resource.data.userId == request.auth.uid;

      match /versions/{versionId} {
        allow read, write: if isAuthenticated()
          && get(/databases/$(database)/documents/nutrition_meals/$(mealId)).data.userId == request.auth.uid;
      }

      // Both household members can log cooks on shared recipes under their own uid
      match /cookLogs/{logId} {
        allow read: if isAuthenticated()
          && (get(/databases/$(database)/documents/nutrition_meals/$(mealId)).data.userId == request.auth.uid
              || get(/databases/$(database)/documents/nutrition_meals/$(mealId)).data.householdShared == true);
        allow create: if isAuthenticated()
          && request.resource.data.userId == request.auth.uid;
        allow update, delete: if isAuthenticated()
          && resource.data.userId == request.auth.uid;
      }
    }

    match /nutrition_plans/{planId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }

    match /nutrition_plan_templates/{templateId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }

    match /nutrition_plan_instances/{instanceId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }

    match /nutrition_plan_confirmations/{docId} {
      allow read, write: if isAuthenticated() && (docId.matches("^" + request.auth.uid + "_.*") || (resource != null && resource.data.userId == request.auth.uid) || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }
    match /nutrition_vitamins/{vitaminId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }
    match /nutrition_vitamin_logs/{logId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }
    match /nutrition_goals/{goalId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }
    match /nutrition_planned_deviations/{deviationId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }
    match /preloadedIngredients/{docId} {
      allow read: if isAuthenticated();
    }
    match /nutrition_weekend_strategies/{strategyId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }
    match /nutrition_deviation_categories/{categoryId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }
    // -------------------------------------------------------------------------
    // Health Sync (Used by Smart Habits)
    // -------------------------------------------------------------------------

    match /health_daily_steps/{docId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create, update: if (isAuthenticated() && request.resource.data.userId == request.auth.uid)
                            || isBridgeAppUser(request.resource.data.userId);
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    match /health_sleep_sessions/{docId} {
      allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create, update: if (isAuthenticated() && request.resource.data.userId == request.auth.uid)
                            || isBridgeAppUser(request.resource.data.userId);
      allow delete: if isAuthenticated() && resource.data.userId == request.auth.uid;
    }
    match /health_sync_metadata/{userId} {
      allow read, write: if (isAuthenticated() && request.auth.uid == userId)
                         || isBridgeAppUser(userId);
    }

    // -------------------------------------------------------------------------
    // Habit Tracker App Collections
    // -------------------------------------------------------------------------

    match /habits/{habitId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }

    match /habit_logs/{logId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }

    match /habit_categories/{categoryId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }

    match /habit_units/{unitId} {
      allow read, write: if isAuthenticated() && (resource == null || resource.data.userId == request.auth.uid || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }

    match /user_preferences/{userId} {
      allow read, write: if isAuthenticated() && (request.auth.uid == userId || isBridgeAppUser(userId));
    }

    // -------------------------------------------------------------------------
    // My Recipe App
    // -------------------------------------------------------------------------

    // Households — two real users sharing a kitchen
    match /households/{householdId} {
      allow read: if request.auth != null;

      allow create: if request.auth != null
        && request.resource.data.ownerId == request.auth.uid
        && request.resource.data.members.size() == 1
        && request.resource.data.members[0] == request.auth.uid;

      // Update path 1: existing member updating the household.
      // Update path 2: a non-member joining via invite code (members grows by 1
      // to include the caller, code stays the same, ownerId unchanged).
      allow update: if request.auth != null && (
        request.auth.uid in resource.data.members
        || (
          resource.data.members.size() < 2
          && request.resource.data.inviteCode == resource.data.inviteCode
          && request.resource.data.ownerId == resource.data.ownerId
          && request.auth.uid in request.resource.data.members
          && request.resource.data.members.size() == resource.data.members.size() + 1
        )
      );

      allow delete: if request.auth != null
        && request.auth.uid == resource.data.ownerId;

      match /pantryState/{docId} {
        allow read, write: if request.auth != null
          && request.auth.uid in get(/databases/$(database)/documents/households/$(householdId)).data.members;
      }
    }

    match /recipes/{recipeId} {
      allow read: if request.auth != null && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
      allow update, delete: if request.auth != null && resource.data.userId == request.auth.uid;

      match /versions/{versionId} {
        allow read, write: if request.auth != null
          && get(/databases/$(database)/documents/recipes/$(recipeId)).data.userId == request.auth.uid;
      }

      match /cookLogs/{logId} {
        allow read, write: if request.auth != null
          && get(/databases/$(database)/documents/recipes/$(recipeId)).data.userId == request.auth.uid;
      }
    }

    match /mealPlans/{planId} {
      allow read: if request.auth != null && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
      allow update, delete: if request.auth != null && resource.data.userId == request.auth.uid;
    }

    match /shoppingLists/{listId} {
      allow read, write: if request.auth != null
        && (resource == null || resource.data.userId == request.auth.uid
            || (request.resource != null && request.resource.data.userId == request.auth.uid));
    }

    // Shopping locations and ingredient categories — per-user docs that
    // Recipe Manager merges across household members client-side.
    // Reads open to any authenticated user; writes remain owner-only.
    match /shoppingLocations/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    match /ingredientCategories/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
